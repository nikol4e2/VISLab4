# Матрица со набљудувани податоци
data <- matrix(c(500, 650, 200, 610, 550, 170), 
               nrow = 2, byrow = TRUE,
               dimnames = list(c("Под 30г.", "Барем 30г."),
                               c("Македонци", "Од ЕУ", "Ниту Македонци, ниту од ЕУ")))

# Преглед на податоците
print("Матрица со набљудувани фреквенции:")
print(data)

# Хи-квадрат тест за независност
chi_sq_test <- chisq.test(data)

# Резултати
cat("Хи-квадрат статистика:", chi_sq_test$statistic, "\n")
cat("P-вредност:", chi_sq_test$p.value, "\n")
cat("Степени на слобода:", chi_sq_test$parameter, "\n")

# Заклучок
if (chi_sq_test$p.value < 0.05) {
  cat("На ниво на значајност од 5%, ги отфрламе H0. Постои зависност меѓу возраста и државата од која доаѓаат патниците.\n")
} else {
  cat("На ниво на значајност од 5%, НЕ можеме да ја отфрлиме H0. Не постои значајна зависност меѓу возраста и државата од која доаѓаат патниците.\n")
}
